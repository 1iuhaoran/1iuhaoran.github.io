整理之前openmp与mpi的代码

openmp

openmp的入门相对比较友好，对于用过thread的人来讲，openmp不算新颖，没有难懂的概念。对于不了解多线程的人，大概也不了解进程，简单来讲，进程就是运行一个程序，线程就是这个程序的执行通道。如何把一个程序分成多个线程，openmp就提供了解决方案。只需要通过两条执行指令，在程序需要执行并行的模块前加入指导语句，在模块结束后填写终止语句。

走进openmp

我们通过下面这个例子走进openmp。求和，通过更改n的值，就可以观察程序运行的结果，注意其时间。在数量级达到百万级，千万级的时候，程序就绷不住了。


#include 
#include 
int main(int argc, char const *argv[])
{
  clock_t start, end;
  double cpu_time_used;
  start = clock();
  int n = 100;
  long long int sum = 0;
  int data[n];
  for (int i = 0; i < n; i++)
  {
    data[i] = i + 1;
  }

  for (int i = 0; i < n; i++)
  {
    sum += data[i];
  }
  printf("sum = %lld\n", sum);
  end = clock();
  cpu_time_used = ((double)(end - start)) / CLOCKS_PER_SEC;
  printf("Time taken: %f\n", cpu_time_used);
  return 0;
}

第一个案例：openmp加速矩阵乘法


#include 
#include 
#include 
#define M 1000
#define N 1000
#define P 1000
int main(int argc, char const *argv[])
{
  //动态分配二维数组a,b,c
  double **a = (double **)malloc(M * sizeof(double *));
  double **b = (double **)malloc(N * sizeof(double *));
  double **c = (double **)malloc(M * sizeof(double *));

  for (int i = 0; i < M; i++)
  {
    a[i] = (double *)malloc(N * sizeof(double));
    c[i] = (double *)malloc(P * sizeof(double));
  }
  for (int i = 0; i < N; i++)
  {
    b[i] = (double *)malloc(P * sizeof(double));
  }
  // 初始化a,b,c...
  // 矩阵乘法 M*N  N*P  M*P
  double start_time = omp_get_wtime();
  #pragma omp parallel for collaspe(2)
  for (int i = 0; i < M; i++)
  {
    for (int j = 0;j < P; j++)
    {
      c[i][j] = 0;
      for (int k = 0;k < N; k++)
      {
        c[i][j] += a[i][k] * b[k][j];
      }
    }
  }
  double end_time = omp_get_wtime();
  printf("运行时间: %f 秒\n", end_time - start_time);
  // 释放分配的内存
  for (int i = 0; i < M; i++)
  {
    free(a[i]);
    free(c[i]);
  }
  for (int j = 0; j < N; j++)
  {
    free(b[j]);
  }
  free(a);
  free(b);
  free(c);

  return 0;

}

opemmp的内置api


#include 
#include 
int main(int argc, char const *argv[])
{
  int tid, ncpu;
  double start_time, end_time;
  start_time = omp_get_wtime();
  tid = omp_get_thread_num();
  ncpu = omp_get_num_threads();
  printf("hello from thread %d in %d CPUs\n", tid, ncpu);
  printf("\n");
#pragma omp parallel num_threads(3) private(tid, ncpu)
  {
    tid = omp_get_thread_num();
    ncpu = omp_get_num_threads();
    printf("hello from thread %d in %d CPUs\n", tid, ncpu);
  }
  printf("\n");
  printf("hello from thread %d in %d Cpus\n", tid, ncpu);
  end_time = omp_get_wtime(); // 结束计时
  // 输出运行时间
  printf("运行时间: %f 秒\n", end_time - start_time);

  return 0;
}

openmp的规约


#include 
#include 
int main(int argc, char const *argv[])
{
  long long int sum = 0;
  int n = 10000000;
  double start_time, end_time;
  start_time = omp_get_wtime();
  int data[100000];
  // 初始化数组
  for (int i = 0; i < n; i++)
  {
    data[i] = i + 1;
  }
#pragma omp parallel for reduction(+ : sum)
  for (int i = 0; i < n; i++)
  {
    sum += data[i];
  }
  printf("%lld\n", sum);
  end_time = omp_get_wtime(); // 结束计时
  // 输出运行时间
  printf("运行时间: %f 秒\n", end_time - start_time);
  return 0;
}

single


#include 
#include 

#define SIZE 10

int main()
{
  int shared_array[SIZE];

#pragma omp parallel

  {
// 使用 single 指令确保只有一个线程初始化数组
#pragma omp single
    {
      printf("线程 %d 开始初始化数组...\n", omp_get_thread_num());

      // 初始化数组
      for (int i = 0; i < SIZE; i++)
      {
        shared_array[i] = i;
      }

      printf("线程 %d 完成初始化数组。\n", omp_get_thread_num());
      int num_threads = omp_get_num_threads();
      printf("当前线程数量：%d\n", num_threads);
    }

// 为了保证数组初始化完成后再进行输出，我们可以使用 barrier
#pragma omp barrier

// 使用互斥锁保护 printf 函数的调用
#pragma omp critical
    {
      printf("线程 %d 访问共享数组：", omp_get_thread_num());
      for (int i = 0; i < SIZE; i++)
      {
        printf("%d ", shared_array[i]);
      }
      printf("\n");
    }
  }
  return 0;
}
