#define M 500
#define N 500
#define P 500
#define BLOCK_SIZE

// 大一新生都会的运算矩阵乘法的函数
void matrixMultiply(double **A, double **B, double **C)
{
  for (size_t i = 0; i < M; i++)
  {
    for (size_t j = 0; j < P; j++)
    {
      C[i][j] = 0;
      for (size_t k = 0; k < N; k++)
      {
        C[i][j] += A[i][k] * B[k][j];
      }
    }
  }
}

// 运用循环重排序/循环交换的矩阵乘法函数
// 优化原因：考虑到一变动k，要跳转指令，吭哧吭哧执行好几条，所以做这种优化
void matrixMultiply_circularshift(double **A, double **B, double **C)
{
  // 初始化C矩阵为零
  for (size_t i = 0; i < M; i++)
  {
    for (size_t j = 0; j < P; j++)
    {
      C[i][j] = 0;
    }
  }

  for (size_t i = 0; i < M; i++)
  {
    for (size_t k = 0; k < N; k++)
    {
      for (size_t j = 0; j < P; j++)
      {
        C[i][j] += A[i][k] * B[k][j];
      }
    }
  }
}

// 循环展开
void matrixMultiply_loopunroll(double **A, double **B, double **C)
{
  for (size_t i = 0; i < M; i++)
  {
    for (size_t j = 0; j < P; j++)
    {
      C[i][j] = 0;
    }
  }

  for (size_t i = 0; i < M; i++)
  {
    for (size_t k = 0; k < N; k++)
    {
      for (size_t j = 0; j < P; j += 4)
      {
        C[i][j] += A[i][k] * B[k][j];
        C[i][j + 1] += A[i][k] * B[k][j + 1];
        C[i][j + 2] += A[i][k] * B[k][j + 2];
        C[i][j + 3] += A[i][k] * B[k][j + 3];
      }
    }
  }
}

// 循环分块
void matrixMultiply_blocked(double **A, double **B, double **C)
{
  size_t i, j, k, J, K;
  size_t T = 4, S = 8;
  for (i = 0; i < N; i++)
  {
    /* code */
    for (j = 0; j < N; j += T)
    {
      /* code */
      for (k = 0; k < N; k += S)
      {
        /* code */
        for (J = j; J < j + T; J++)
        {
          for (K = k; K < k + S; K++)
          {
            C[i][K] += A[i][J] * B[J][K];
          }
        }
      }
    }
  }
}

void matrixMultiply_vectorized(double **A, double **B, double **C)
{
  __m256d xmm_a, xmm_b, xmm_c;
  double sum[4];

  // 初始化矩阵为0
  for (size_t i = 0; i < M; i++)
  {
    /* code */
    for (size_t j = 0; j < P; j++)
    {
      /* code */
      C[i][j] = 0;
    }

    // 矩阵乘法
    for (size_t i = 0; i < M; i++)
    {
      /* code */
      for (size_t k = 0; k < N; k++)
      {
        /* code */
        for (size_t j = 0; j < P; j++)
        {
          /* code */
          // 加载A矩阵的4个元素到向量寄存器
          xmm_a = _mm256_set_pd(A[i][k], A[i][k], A[i][k], A[i][k]);

          // 加载B矩阵的4个元素到向量寄存器
          xmm_b = _mm256_loadu_pd(&B[k][j]);

          // 加载C矩阵的当前值
          xmm_c = _mm256_loadu_pd(&C[i][j]);

          // 执行并行乘法并累加
          xmm_c = _mm256_fmadd_pd(xmm_a, xmm_b, xmm_c);

          // 存储结果回C矩阵
          _mm256_storeu_pd(&C[i][j], xmm_c);
        }
      }
    }
  }
}

// 检查矩阵乘积结果是否正确的函数
int checkResult(double **A, double **B, double **C)
{
  for (size_t i = 0; i < M; i++)
  {
    for (size_t j = 0; j < P; j++)
    {
      double sum = 0;
      for (size_t k = 0; k < N; k++)
      {
        sum += A[i][k] * B[k][j];
      }
      if (abs(C[i][j] - sum) > 1e-6)
      {
        printf("Mismatch at C[%zu][%zu]: expected %f, got %f\n", i, j, sum, C[i][j]);
        return 0;
      }
    }
  }
  return 1;
}

int main(int argc, char const *argv[])
{
  double **A = (double **)malloc(M * sizeof(double *));
  double **B = (double **)malloc(N * sizeof(double *));
  double **C = (double **)malloc(M * sizeof(double *));

  for (size_t i = 0; i < M; i++)
  {
    A[i] = (double *)malloc(N * sizeof(double));
    C[i] = (double *)malloc(P * sizeof(double));
  }
  for (size_t j = 0; j < N; j++)
  {
    /* code */
    B[j] = (double *)malloc(P * sizeof(double));
  }

  // 初始化随机数生成器
  srand(time(0));

  // 初始化矩阵A和矩阵B为随机值
  for (size_t i = 0; i < M; i++)
  {
    /* code */
    for (size_t j = 0; j < N; j++)
    {
      /* code */
      A[i][j] = (double)rand() / RAND_MAX;
    }
  }

  for (size_t i = 0; i < N; i++)
  {
    /* code */
    for (size_t j = 0; j < P; j++)
    {
      /* code */
      B[i][j] = (double)rand() / RAND_MAX;
    }
  }

  // 记录开始时间
  clock_t start = clock();

  // 计算结果
  // matrixMultiply(A, B, C);
  // matrixMultiply_circularshift(A, B, C);
  // matrixMultiply_loopunroll(A, B, C);
  // matrixMultiply_blocked(A, B, C);
  matrixMultiply_vectorized(A, B, C);

  // 记录结束时间
  clock_t end = clock();
  double time_spent = (double)(end - start) / CLOCKS_PER_SEC;

  printf("Matrix multiplication completed in %f seconds.\n", time_spent);

  // 检查结果
  if (checkResult(A, B, C))
  {
    printf("Matrix multiplication result is correct.\n");
  }
  else
  {
    printf("Matrix multiplication result is incorrect.\n");
  }

  // 释放内存
  for (size_t i = 0; i < M; i++)
  {
    free(A[i]);
    free(C[i]);
  }
  for (size_t j = 0; j < N; j++)
  {
    free(B[j]);
  }
  free(A);
  free(B);
  free(C);

  return 0;
}
