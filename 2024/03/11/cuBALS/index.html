cuBALS

cuBALS体验

cuBLAS 是 NVIDIA CUDA 提供的一套 BLAS（基本线性代数子程序）库，专门用于在 NVIDIA 的 GPU 上高效地执行线性代数运算。下面是一个使用 cuBLAS 执行矩阵乘法的基本示例步骤：


#include 
#include 
#include 

int main() {
    // 定义矩阵维度
    int n = 3;
    float *A, *B, *C;
    float *d_A, *d_B, *d_C;

    // 分配主机内存
    A = (float *)malloc(n * n * sizeof(float));
    B = (float *)malloc(n * n * sizeof(float));
    C = (float *)malloc(n * n * sizeof(float));

    // 初始化矩阵A和B
    for(int i = 0; i < n * n; i++) {
        A[i] = float(i);
        B[i] = float(i);
    }

    // 分配设备内存
    cudaMalloc((void **)&d_A, n * n * sizeof(float));
    cudaMalloc((void **)&d_B, n * n * sizeof(float));
    cudaMalloc((void **)&d_C, n * n * sizeof(float));

    // 创建一个cuBLAS上下文
    cublasHandle_t handle;
    cublasCreate(&handle);

    // 将矩阵数据从主机复制到设备
    cublasSetMatrix(n, n, sizeof(float), A, n, d_A, n);
    cublasSetMatrix(n, n, sizeof(float), B, n, d_B, n);

    float alpha = 1.0f;
    float beta = 0.0f;

    // 执行矩阵乘法 C = alpha * A * B + beta * C
    // 注意：cuBLAS是列主序的
    cublasSgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, n, n, n, &alpha, d_A, n, d_B, n, &beta, d_C, n);

    // 将结果从设备复制回主机
    cublasGetMatrix(n, n, sizeof(float), d_C, n, C, n);

    // 打印结果矩阵C
    printf("Result matrix C:\n");
    for(int i = 0; i < n; i++) {
        for(int j = 0; j < n; j++) {
            printf("%f ", C[i * n + j]);
        }
        printf("\n");
    }

    // 释放资源
    cudaFree(d_A);
    cudaFree(d_B);
    cudaFree(d_C);
    free(A);
    free(B);
    free(C);
    cublasDestroy(handle);

    return 0;
}
