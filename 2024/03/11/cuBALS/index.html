cuBALS

在做完线程组织和共享内存优化后，4000维度的矩阵运算能达到100ms以内，再进行下一步优化之前，不妨体验一下英伟达自己的矩阵运算函数

cuBALS体验

cuBLAS 是 NVIDIA CUDA 提供的一套 BLAS（基本线性代数子程序）库，专门用于在 NVIDIA 的 GPU 上高效地执行线性代数运算。下面是一个使用 cuBLAS 执行矩阵乘法的基本示例步骤：


#include < stdio.h >
#include < stdlib.h >
#include < cuda_runtime.h >
#include < cublas_v2.h >

#define WIDTH 4000

void initializeMatrix(float *mat, int width)
{
  for (int i = 0; i < width * width; i++)
  {
    mat[i] = rand() % 100 / 10.0f; // 随机初始化矩阵元素
  }
}

int main()
{
  float *A, *B, *C;
  float *d_A, *d_B, *d_C;
  int size = WIDTH * WIDTH * sizeof(float);

  // 分配主机内存
  A = (float *)malloc(size);
  B = (float *)malloc(size);
  C = (float *)malloc(size);

  // 初始化矩阵A和B
  initializeMatrix(A, WIDTH);
  initializeMatrix(B, WIDTH);

  // 创建和初始化CUBLAS库对象
  cublasHandle_t handle;
  cublasCreate(&handle);

  // 分配设备内存
  cudaMalloc((void **)&d_A, size);
  cudaMalloc((void **)&d_B, size);
  cudaMalloc((void **)&d_C, size);

  // 将矩阵A和B复制到设备内存
  cublasSetMatrix(WIDTH, WIDTH, sizeof(float), A, WIDTH, d_A, WIDTH);
  cublasSetMatrix(WIDTH, WIDTH, sizeof(float), B, WIDTH, d_B, WIDTH);

  float alpha = 1.0f;
  float beta = 0.0f;

  // 创建CUDA事件进行计时
  cudaEvent_t start, stop;
  cudaEventCreate(&start);
  cudaEventCreate(&stop);

  // 记录开始时间
  cudaEventRecord(start, 0);

  // 使用CUBLAS进行矩阵乘法 d_C = alpha * d_A * d_B + beta * d_C
  // 注意：cuBLAS是列优先的，所以参数是以列优先的方式传递的
  cublasSgemm(handle, CUBLAS_OP_N, CUBLAS_OP_N, WIDTH, WIDTH, WIDTH, &alpha, d_A, WIDTH, d_B, WIDTH, &beta, d_C, WIDTH);

  // 记录结束时间
  cudaEventRecord(stop, 0);
  cudaEventSynchronize(stop);

  // 计算并打印执行时间
  float milliseconds = 0;
  cudaEventElapsedTime(&milliseconds, start, stop);
  printf("Matrix multiplication took %f milliseconds\n", milliseconds);

  // 将结果矩阵C复制回主机内存
  cublasGetMatrix(WIDTH, WIDTH, sizeof(float), d_C, WIDTH, C, WIDTH);

  // 清理资源
  cudaFree(d_A);
  cudaFree(d_B);
  cudaFree(d_C);
  cublasDestroy(handle);

  free(A);
  free(B);
  free(C);

  return 0;
}
